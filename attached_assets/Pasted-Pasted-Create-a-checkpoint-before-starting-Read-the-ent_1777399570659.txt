TASK 1 — PRO GATE SYSTEM
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Create a single source of truth for Pro status.
Add this to the top of the main JS block in 
artifacts/promptmegood/index.html:

const PMG_PRO_KEY = 'promptmegood:pro:v1';

function pmgIsPro() {
  try {
    return localStorage.getItem(PMG_PRO_KEY) 
      === 'true';
  } catch(e) { return false; }
}

function pmgUnlockPro() {
  try {
    localStorage.setItem(PMG_PRO_KEY, 'true');
    document.body.classList.add('pmg-is-pro');
    document.querySelectorAll(
      '.pmg-pro-feature'
    ).forEach(function(el) {
      el.classList.remove('pmg-locked');
    });
    showProUnlockedToast();
  } catch(e) {}
}

function showProUnlockedToast() {
  const t = document.createElement('div');
  t.className = 'pmg-pro-toast';
  t.innerHTML = '🎉 Pro Unlocked! All features 
    are now available.';
  document.body.appendChild(t);
  setTimeout(function() { t.remove(); }, 4000);
}

On page load check pro status:
if (pmgIsPro()) {
  document.body.classList.add('pmg-is-pro');
}

TO UNLOCK PRO (you call this from browser 
console when a user pays):
pmgUnlockPro()

TO REMOVE PRO (for testing):
localStorage.removeItem('promptmegood:pro:v1')
then refresh.
