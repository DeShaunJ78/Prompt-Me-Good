async function generatePrompt() {
  // Collect all builder inputs
  const payload = {
    goal: document.getElementById('goal').value.trim(),
    category: document.getElementById('category').value,
    experience: document.getElementById('experience').value,
    tone: document.getElementById('tone').value,

  };

  // Validate goal field

    showToast('Please enter a goal before generating.');
    return;
  }

  // Set loading state
  setGenerateButtonLoading(true);
  setOutputPlaceholder('Generating your prompt…');

  try {
    const response = await fetch('/api/generate', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(payload)
    });

    const data = await response.json();

    if (data.prompt) {
      // Display generated prompt in output panel
      setOutputContent(data.prompt);
      scrollToOutput();
      showToast('Prompt generated.');
      saveToHistory(data.prompt, payload);
    } else {
      setOutputPlaceholder('Something went wrong. Please try again.');
      showToast('Generation failed — please try again.');
      console.error('API error:', data.error);
    }

  } catch (err) {
    setOutputPlaceholder('Connection error. Please check your internet and try again.');
    showToast('Could not reach the server.');
    console.error('Fetch error:', err);
  } finally {
    setGenerateButtonLoading(false);
  }
}

Add a /api/generate-stream endpoint that streams the OpenAI response using Server-Sent Events:

app.post('/api/generate-stream', async (req, res) => {
  res.setHeader('Content-Type', 'text/event-stream');
  res.setHeader('Cache-Control', 'no-cache');
  res.setHeader('Connection', 'keep-alive');

  try {
    const stream = await openai.chat.completions.create({
      model: 'gpt-4o',
      stream: true,
      messages: buildMessages(req.body) // reuse existing message builder
    });

    for await (const chunk of stream) {
      const text = chunk.choices[0]?.delta?.content || '';
      if (text) {
        res.write(`data: ${JSON.stringify({ text })}\n\n`);
      }
    }

    res.write('data: [DONE]\n\n');
    res.end();

  } catch (err) {
    res.write(`data: ${JSON.stringify({ error: err.message })}\n\n`);
    res.end();
  }
});

async function generatePromptStream() {
  const payload = collectBuilderInputs(); // reuse existing input collector

  if (!payload.goal) {
    showToast('Please enter a goal before generating.');
    return;
  }

  setGenerateButtonLoading(true);
  setOutputContent(''); // clear output
  setOutputPlaceholder('');

  let fullText = '';

  try {
    const response = await fetch('/api/generate-stream', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(payload)
    });

    const reader = response.body.getReader();
    const decoder = new TextDecoder();

    scrollToOutput(); // scroll immediately so user sees text appear

    while (true) {
      const { done, value } = await reader.read();
      if (done) break;

      const lines = decoder.decode(value).split('\n');
      for (const line of lines) {
        if (line.startsWith('data: ')) {
          const raw = line.slice(6);
          if (raw === '[DONE]') {
            showToast('Prompt generated.');
            saveToHistory(fullText, payload);
            break;
          }
          try {
            const parsed = JSON.parse(raw);
            if (parsed.text) {
              fullText += parsed.text;
              appendOutputContent(parsed.text); // append word by word
            }
          } catch (_) {}
        }
      }
    }

  } catch (err) {
    setOutputPlaceholder('Connection error. Please try again.');
    showToast('Could not reach the server.');
    console.error('Stream error:', err);
  } finally {
    setGenerateButtonLoading(false);
  }
}
