## What Changes

The hero section of `index.html` is being rewritten to clarify the product's function and add an interactive mini-generator.

1. **Headline & Lede Copy:** Updated to focus on function rather than feeling.
2. **Meta Row Repositioning:** The trust signals ("Free tier", "No signup", "Local-first") move *above* the CTA buttons.
3. **Mini-Generator:** A textarea and a "Build My Prompt →" button are added to the hero.
4. **Pre-Baked Demo:** When the user clicks "Build My Prompt →", a pre-baked structured prompt types out below the buttons.
5. **Post-Output CTA:** A new CTA appears below the typed output, driving users to the full app.

## 3. JavaScript Implementation (Pre-Baked Demo)

Add the following script block at the end of `index.html`, just before `</body>`:

```html
<script>
document.addEventListener('DOMContentLoaded', function() {
  const input = document.getElementById('hero-input');
  const buildBtn = document.getElementById('hero-build-btn');
  const outputContainer = document.getElementById('hero-output-container');
  const output = document.getElementById('hero-output');
  
  // Pre-baked demo responses
  const demos = [
    {
      keywords: ['email', 'sales', 'pitch', 'outreach'],
      text: '<b>Role:</b> Senior B2B sales copywriter\n<b>Audience:</b> Operations leads at 50–200 person SaaS companies\n<b>Goal:</b> Book a 15-minute discovery call\n<b>Tone:</b> Direct, friendly, zero fluff\n<b>Constraints:</b> 90–120 words, plain text, one CTA\n<b>Format:</b> Subject line + 3 short paragraphs + CTA\n<b>CTA:</b> "Grab a 15-min slot →" — link to a Calendly URL'
    },
    {
      keywords: ['blog', 'post', 'article', 'seo'],
      text: '<b>Role:</b> Expert content marketer and SEO specialist\n<b>Audience:</b> Marketing managers looking for actionable growth tactics\n<b>Goal:</b> Drive newsletter signups through high-value educational content\n<b>Tone:</b> Authoritative, tactical, engaging\n<b>Constraints:</b> ~1200 words, short paragraphs, highly skimmable\n<b>Format:</b> Hook intro + 3 actionable steps with examples + conclusion\n<b>CTA:</b> "Subscribe for weekly tactics"'
    },
    {
      keywords: ['code', 'script', 'function', 'bug', 'fix'],
      text: '<b>Role:</b> Senior Staff Software Engineer\n<b>Audience:</b> Mid-level developer needing a robust, production-ready solution\n<b>Goal:</b> Provide a clean, optimized, and well-documented code solution\n<b>Tone:</b> Technical, precise, helpful\n<b>Constraints:</b> Include error handling, edge cases, and inline comments\n<b>Format:</b> Brief explanation + code block + usage example\n<b>Context:</b> Prioritize readability and performance'
    },
    {
      keywords: ['default'], // Fallback
      text: '<b>Role:</b> Expert consultant in the relevant field\n<b>Audience:</b> Professionals seeking clear, actionable advice\n<b>Goal:</b> Provide a comprehensive and structured response\n<b>Tone:</b> Professional, clear, concise\n<b>Constraints:</b> No fluff, highly structured, easy to read\n<b>Format:</b> Executive summary + bulleted key points + next steps'
    }
  ];

  function getDemoResponse(text) {
    const lowerText = text.toLowerCase();
    for (const demo of demos) {
      if (demo.keywords.includes('default')) return demo.text;
      if (demo.keywords.some(kw => lowerText.includes(kw))) return demo.text;
    }
    return demos[demos.length - 1].text;
  }

  function typeOut(element, html, speed = 15) {
    element.innerHTML = '';
    let i = 0;
    let isTag = false;
    let currentHTML = '';
    
    function type() {
      if (i < html.length) {
        currentHTML += html.charAt(i);
        element.innerHTML = currentHTML;
        
        if (html.charAt(i) === '<') isTag = true;
        if (html.charAt(i) === '>') isTag = false;
        
        i++;
        setTimeout(type, isTag ? 0 : speed);
      } else {
        // Ensure mobile scroll into view once typing finishes
        if (window.innerWidth <= 767) {
           outputContainer.scrollIntoView({ behavior: 'smooth', block: 'nearest' });
        }
      }
    }
    type();
  }

  buildBtn.addEventListener('click', function() {
    const val = input.value.trim();
    if (!val) {
      input.focus();
      return;
    }
    
    const responseHTML = getDemoResponse(val);
    outputContainer.hidden = false;
    buildBtn.disabled = true;
    buildBtn.textContent = 'Building...';
    
    // Simulate slight network delay for realism
    setTimeout(() => {
      buildBtn.textContent = 'Prompt Built ✓';
      typeOut(output, responseHTML);
    }, 400);
  });
});
</script>
```
