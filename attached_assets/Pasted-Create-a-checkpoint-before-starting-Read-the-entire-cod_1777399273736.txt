Create a single source of truth for Pro status.
Add this to the top of the main JS block in 
artifacts/promptmegood/index.html:

const PMG_PRO_KEY = 'promptmegood:pro:v1';

function pmgIsPro() {
  try {
    return localStorage.getItem(PMG_PRO_KEY) 
      === 'true';
  } catch(e) { return false; }
}

function pmgUnlockPro() {
  try {
    localStorage.setItem(PMG_PRO_KEY, 'true');
    document.body.classList.add('pmg-is-pro');
    document.querySelectorAll(
      '.pmg-pro-feature'
    ).forEach(function(el) {
      el.classList.remove('pmg-locked');
    });
    showProUnlockedToast();
  } catch(e) {}
}

function showProUnlockedToast() {
  const t = document.createElement('div');
  t.className = 'pmg-pro-toast';
  t.innerHTML = '🎉 Pro Unlocked! All features 
    are now available.';
  document.body.appendChild(t);
  setTimeout(function() { t.remove(); }, 4000);
}

On page load check pro status:
if (pmgIsPro()) {
  document.body.classList.add('pmg-is-pro');
}

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Add this JavaScript function to index.html:

function showUpgradeModal(featureName) {
  const existing = document.getElementById(
    'pmg-upgrade-overlay'
  );
  if (existing) existing.remove();

  const overlay = document.createElement('div');
  overlay.className = 'pmg-upgrade-overlay';
  overlay.id = 'pmg-upgrade-overlay';

  overlay.innerHTML = 
    '<div class="pmg-upgrade-modal">' +
    '<span class="pmg-upgrade-modal-icon">' +
      '🔒</span>' +
    '<h3>' + featureName + ' Is A Pro Feature' +
      '</h3>' +
    '<p>Upgrade to Pro for $9/month and unlock ' +
      'unlimited access to this feature plus ' +
      'cloud sync, image generation, file ' +
      'analysis, and more.</p>' +
    '<div class="pmg-upgrade-modal-actions">' +
    '<a class="pmg-upgrade-cta" ' +
      'href="./pricing.html#early-access">' +
      'Join Pro Early Access</a>' +
    '<button class="pmg-upgrade-dismiss" ' +
      'id="pmg-upgrade-dismiss-btn">' +
      'Maybe Later</button>' +
    '</div></div>';

  document.body.appendChild(overlay);

  document.getElementById(
    'pmg-upgrade-dismiss-btn'
  ).addEventListener('click', function() {
    overlay.remove();
  });

  overlay.addEventListener('click', 
    function(e) {
      if (e.target === overlay) overlay.remove();
    }
  );
}
