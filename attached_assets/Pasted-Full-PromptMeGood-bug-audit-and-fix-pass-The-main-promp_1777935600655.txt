Accessibility/DOM:
Run hidden focusable scan on mobile and desktop.
Target: 0–5 hidden focusables maximum.

Use this scan:
```js
[...document.querySelectorAll('a[href], button, input, textarea, select, details, [tabindex]:not([tabindex="-1"]), [role="button"], [contenteditable="true"]')]
  .filter(el => !el.disabled && el.getAttribute('aria-disabled') !== 'true' && el.getAttribute('tabindex') !== '-1')
  .filter(el => {
    const r = el.getBoundingClientRect();
    const cs = getComputedStyle(el);
    return (r.width === 0 || r.height === 0 || cs.display === 'none' || cs.visibility === 'hidden' || el.offsetParent === null)
      && !el.closest('[hidden]')
      && !el.closest('[inert]');
  }).length
